describe('Automation for some links on the-internet.herokuapp.com', () => {
  beforeEach(() => {
    cy.visit('https://the-internet.herokuapp.com');
  });

  it('Logs in with valid credentials', () => {
    cy.contains('Form Authentication').click();
    cy.get('#username').type('tomsmith');
    cy.get('#password').type('SuperSecretPassword!');
    cy.get('.radius').click();
    cy.url().should('include', '/secure');
  });

  it('Enters a number in the input box', () => {
    cy.contains('Inputs').click();
    cy.get('input[type="number"]').type(42);
    cy.get('input[type="number"]').should('have.value', '42');
  });

  it('Hover over user avatars to see profile links', () => {
    cy.contains('Hovers').click();
    cy.get('.figure').eq(0).trigger('mouseover');
    cy.get('.figure').eq(0).find('a').should('be.visible');
    cy.get('.figure').eq(1).trigger('mouseover');
    cy.get('.figure').eq(1).find('a').should('be.visible');
    cy.get('.figure').eq(2).trigger('mouseover');
    cy.get('.figure').eq(2).find('a').should('be.visible');
  });

  it('Selects an option from the dropdown', () => {
    cy.contains('Dropdown').click();
    cy.get('#dropdown').select('Option 2');
    cy.get('#dropdown').should('have.value', '2');
  });
});
